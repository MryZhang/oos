Include object.hpp 

#include "object/object.hpp"

Create a class which is derived from oos::object. 

class Artist : public oos::object
{

Add your private members for the class 

private:
  std::string name_;

Add your constructors including a default constructor 

public:
  Artist();
  Artist(const std::string &n);
  virtual ~Artist();

Add the serialization interface to read and write your members 

  void read_from(oos::object_atomizer *reader);
  void write_to(oos::object_atomizer *writer) const;

Add access and modifier methods for your members 

  std::string name() const;
  void name(const std::string &n);

Add all methods you would like to have in your class. In this example its an std::ostream operator 

  friend std::ostream& operator <<(std::ostream &os, const Artist &a);

That's all for your header 

};
